DROP TABLE IF EXISTS `weibo_user`;
CREATE TABLE `weibo_user` (
  `id` varchar(32) COLLATE utf8_bin NOT NULL,
  `screenName` varchar(20) COLLATE utf8_bin DEFAULT NULL,
  `name` varchar(20) COLLATE utf8_bin DEFAULT NULL,
  `province` int(11) DEFAULT NULL,
  `city` int(11) DEFAULT NULL,
  `location` varchar(20) COLLATE utf8_bin DEFAULT NULL,
  `description` varchar(30) COLLATE utf8_bin DEFAULT NULL,
  `url` varchar(30) COLLATE utf8_bin DEFAULT NULL,
  `profile_image_url` text COLLATE utf8_bin,
  `user_domain` varchar(30) COLLATE utf8_bin DEFAULT NULL,
  `gender` char(1) COLLATE utf8_bin DEFAULT NULL,
  `followers_count` int(11) DEFAULT NULL,
  `friends_count` int(11) DEFAULT NULL,
  `favourites_count` int(11) DEFAULT NULL,
  `created_at` datetime DEFAULT NULL,
  `following` tinyint(11) DEFAULT NULL,
  `verified` int(11) DEFAULT NULL,
  `verified_type` int(11) DEFAULT NULL,
  `allow_all_act_msg` tinyint(11) DEFAULT NULL,
  `allow_all_comment` tinyint(11) DEFAULT NULL,
  `follow_me` int(11) DEFAULT NULL,
  `avatar_large` text COLLATE utf8_bin,
  `online_status` int(11) DEFAULT NULL,
  `bi_followers_count` int(11) DEFAULT NULL,
  `remark` varchar(255) COLLATE utf8_bin DEFAULT NULL,
  `lang` varchar(10) COLLATE utf8_bin DEFAULT NULL,
  `verified_reason` varchar(30) COLLATE utf8_bin DEFAULT NULL,
  `weihao` varchar(30) COLLATE utf8_bin DEFAULT NULL,
  `status_Id` varchar(16) COLLATE utf8_bin DEFAULT NULL,
  `create_date` datetime DEFAULT NULL COMMENT '创建时间',
  `create_by` varchar(32) COLLATE utf8_bin DEFAULT NULL COMMENT '创建人',
  `update_date` datetime DEFAULT NULL COMMENT '修改时间',
  `update_by` varchar(32) COLLATE utf8_bin DEFAULT NULL COMMENT '修改人',
  `del_flag` char(1) COLLATE utf8_bin DEFAULT '0' COMMENT '删除标记',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_bin;
